const fs = require("fs"); const path = require("path"); // =============================== // CONFIG // =============================== const ARTICLES_DIR = path.join(__dirname, "artilces"); const INDEX_FILE = path.join(ARTICLES_DIR, "index.html"); // =============================== // SAFETY CHECKS // =============================== if (!fs.existsSync(ARTICLES_DIR)) { throw new Error("artilces folder not found"); } if (!fs.existsSync(INDEX_FILE)) { throw new Error("artilces/index.html not found"); } // =============================== // READ INDEX FILE // =============================== const html = fs.readFileSync(INDEX_FILE, "utf8"); // =============================== // READ ARTICLE FILES // =============================== const articles = fs.readdirSync(ARTICLES_DIR) .filter(file => file.endsWith(".html") && file !== "index.html") .sort(); // =============================== // BUILD ARTICLE LIST // =============================== const list = articles.map(file => { const slug = file.replace(".html", ""); const title = slug .replace(/-/g, " ") .replace(/\b\w/g, c => c.toUpperCase()); return `
  • ${title}
    • `; }).join("\n"); // =============================== // REPLACE LIST BLOCK // =============================== const updated = html.replace( /[\s\S]*?/, `\n
      \n${list}\n
    \n` ); // =============================== // WRITE BACK // =============================== fs.writeFileSync(INDEX_FILE, updated, "utf8"); // =============================== // DONE // =============================== console.log("✅ Articles index updated: /artilces/index.html");
